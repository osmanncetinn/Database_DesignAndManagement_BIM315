SELECT	 COUNT([TOTAL INVOICES])
		,SUM([TOTAL SALES]) AS [TOTAL SALES]
		,MIN([MINIMUM SALES]) as min
		,MAX([MAXIMUM SALES]) as max
		,AVG([AVARAGE SALES])  as avg
FROM (
		SELECT	 COUNT(DISTINCT INVOICE.INV_NUMBER) AS [TOTAL INVOICES]
				,SUM(LINE_UNITS*LINE.LINE_PRICE) AS [TOTAL SALES]
				,MIN(LINE_UNITS*LINE.LINE_PRICE) AS [MINIMUM SALES]
				,MAX(LINE_UNITS*LINE.LINE_PRICE) AS [MAXIMUM SALES]
				,AVG(LINE_UNITS*LINE.LINE_PRICE) AS [AVARAGE SALES]
		FROM INVOICE
		INNER JOIN LINE ON INVOICE.INV_NUMBER=LINE.INV_NUMBER
		GROUP BY INVOICE.INV_NUMBER
	 ) AS TABLE1
